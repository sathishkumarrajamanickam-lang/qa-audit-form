Follow these steps once. All 3 leads share the same HTML file — audits from all of them go into one Google Sheet automatically.

1
Create a Google Sheet
Go to sheets.new and create a new sheet. Name it QA Audits. Share it with all 3 leads (Editor access).
2
Open Apps Script
In your Google Sheet → click Extensions → Apps Script. Delete any existing code.
3
Paste this code
Copy the entire block below and paste it into the Apps Script editor: 
const SHEET_NAME = 'Audits';

function doGet(e) {
  try {
    var p        = e.parameter;
    var callback = p.cb || 'qaGASCallback';

    // Health-check ping (no auditor field)
    if (!p.auditor) {
      return ContentService
        .createTextOutput(callback + '({"status":"ready"})')
        .setMimeType(ContentService.MimeType.JAVASCRIPT);
    }

    var ss    = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName(SHEET_NAME);

    if (!sheet) {
      sheet = ss.insertSheet(SHEET_NAME);
      var headers = [
        'Timestamp','Date','Auditor','Agent','Source','Reference ID',
        'Score','Result',
        'P1','P2','P3','P4','P5','P6','P7','P8','P9',
        'P1 Name','P2 Name','P3 Name','P4 Name','P5 Name',
        'P6 Name','P7 Name','P8 Name','P9 Name',
        'Strengths','Improvement Areas','Feedback'
      ];
      sheet.appendRow(headers);
      sheet.getRange(1, 1, 1, headers.length)
           .setFontWeight('bold')
           .setBackground('#1F2D3D')
           .setFontColor('#FFFFFF');
      sheet.setFrozenRows(1);
    }

    sheet.appendRow([
      new Date().toISOString(),
      p.date    ||'', p.auditor||'', p.agent  ||'',
      p.source  ||'', p.ref    ||'',
      p.score   ||'', p.result ||'',
      p.p0||'', p.p1||'', p.p2||'',
      p.p3||'', p.p4||'', p.p5||'',
      p.p6||'', p.p7||'', p.p8||'',
      p.pn0||'', p.pn1||'', p.pn2||'',
      p.pn3||'', p.pn4||'', p.pn5||'',
      p.pn6||'', p.pn7||'', p.pn8||'',
      p.strength||'', p.improve||'', p.feedback||''
    ]);

    var okJson = JSON.stringify({status:'ok'});
    return ContentService
      .createTextOutput(callback + '(' + okJson + ')')
      .setMimeType(ContentService.MimeType.JAVASCRIPT);

  } catch(err) {
    var cb2 = (e.parameter.cb || 'qaGASCallback');
    return ContentService
      .createTextOutput(cb2 + '({"status":"error","message":' + JSON.stringify(err.message) + '})')
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }
}
